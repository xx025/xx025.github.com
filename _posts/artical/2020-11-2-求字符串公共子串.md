

``` js
/*
需求：
注意：空格只是为了易于观察实际与空格出现与否无关
s1='大狗我爱你    小明非常    的爱'
s2='    我爱你的家小明非常很是的爱'
fun(s1,s2)=>[ '我爱你', '小明非常', '的爱' ]
*/

s1 = '大狗我爱你    小明非常    的爱'
s2 = '    我爱你的家小明非常很是的爱'

function fun1(s1, s2) {
    m = s1.length
    n = s2.length
    sx = []; //存储差异矩阵
    sr = []; //存储全部子串
    sssr = []; //临时存储子串
    for (let i = 0; i < m; i++) {
        sx[i] = [];
        for (let j = 0; j < n; j++) {
            if (s1.charAt(i) == s2.charAt(j)) {
                sx[i][j] = 1;
            } else {
                sx[i][j] = 0;
            }
        }
    }
    k = 0, l = 0, pi = 0
    while (k < m) {
        l = pi;
        while (l < n) {
            sssr = []
            while (l < n && sx[k][l] == 1) {
                sssr.push(s1.charAt(k))
                k++;
                l++;
                pi = l; //记录位置
            }
            if (sssr.length > 0) {
                sr.push(sssr.join(""))
            }
            l++; //前移一位
        }
        k++; //扫描完了一行，换下一行
    }
    return sr;
}
console.log(fun1(s1, s2))
```

输出：

``` 
[ '我爱你', '小明非常', '的爱' ]
```
