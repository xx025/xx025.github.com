## C:

``` C
#include <stdio.h>
void main()
{
    int k = 1;
    for (size_t i = 1; i <= 10; i++)
    {
        k *= 2;
    }

    printf("%d", k);
}
```
## C++
```c++
#include <iostream>
using namespace std;
int main()
{

    int k = 1;
    for (size_t i = 0; i < 10; i++, k *= 2)
        ;
    cout << k << endl;
    return 0;
}

```

## Java
```java
public class main {
    public static void main(String[] args) {
        int k = 1;
        for (int i = 0; i < 10; i++) {
            k *= 2;
        }
        System.out.println(k);
    }
}
```
